function fetch() {
    LightBulb.users.getAlbums({user:"me"}, function (response) {
        console.log(response);
        $("#result").html("<h2>Result</h2>")
        for (i in response.data) {
            var album = response.data[i];
            $("#result").append("<b>" + i + ":</b> " + album.name + "<br/>");
            //$("#result").append("<br/><br/>")
        }
    });
}
