Fetch user activities 

LightBulb.users.getActivities({user:"me"}, function(response) {
    //console.log(response);
    $("#result").html("<h2>Result</h2>")
    for(i in response.data){
        var activity = response.data[i];
        //console.log(activity);
        for(j in activity)
            $("#result").append("<b>"+j+":</b> "+activity[j]+"<br/>");
        $("#result").append("<br/><br/>")
    }
});
